JavaScript Code Snippet : 
    $scope.offChange = function () {
        if ($scope.officemodel == "")
            $scope.depts = null;
        if ($scope.officemodel == "BRA")
            $scope.depts = braDept;
        if ($scope.officemodel == "EDIC")
            $scope.depts = edicDept;
        if ($scope.officemodel == "UNION")
            $scope.depts = unionDept;
    }
