HTML Code Snippet : 
<li ng-class="{active:x.active}" ng-click="toggle(x)" ng-repeat="x in courses">
        {{ x.cn }} : ${{ x.cost }}
</li>
        Total Cost : ${{ total() }}


JavaScript Code Snippet : 
    function coursesctrl($scope) {
        $scope.toggle = function (s) {
            s.active = !s.active;
        };

        $scope.total = function () {
            var total = 0;
            angular.forEach($scope.courses, function (s) {
                if (s.active) {
                    total += s.cost;
                }
            });
            return total;
        };
    }
    angular.module('myapp', []).controller('coursesctrl', coursesctrl)


About the Experiment : 

This experiment is about selecting and dropping courses. We use toggle() function to identify the alternate clicks. ng-class is used to provide the CSS.
